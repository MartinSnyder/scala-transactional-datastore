Our data set

case class Person(
  givenName: String,
  familyName: String,
  birthday: LocalDate,
  occupation: Option[String]) extends Record

val dataStore = new InMemoryDataStore(Seq(
  UniqueConstraint(classOf[Person].getName, "givenName")))

dataStore...createRecords(Seq(
  Person("Abe", "Allen", LocalDate.of(1960, 1, 1),
    None),
  Person("Betsy", "Baker", LocalDate.of(1970, 2, 2),
    Some("Accountant"))
))

dataStore...createRecords((1 to 1000000)
  .map(number => Person(s"Bee #$number", "Worker", LocalDate.now,
    Some("Busywork")))
)

Example 1 - Scala Predicates

connection
  .filter[Person](
    person => person.occupation.isEmpty
  )
>> 116 mSecs: Success(List(Abe Allen))

connection
  .filter[Person](
    person => person.givenName == "Bee #91236"
  )
>> 102 mSecs: Success(List(Bee #91236 Worker))

Example 2 - "Conditions" API

connection
  .retrieveRecords[Person](
    EqualsCondition("occupation", None)
  )
>> 490 mSecs: Success(List(Abe Allen))

connection
  .retrieveRecords[Person](
    EqualsCondition("givenName", "Bee #91236")
  )
>> 2 mSecs: Success(List(Bee #91236 Worker))

Example 3 - Quill

connection
  .retrieveRecords[Person](
    quote { query[Person].filter(_.occupation == None) }
  )
>> 810 mSecs: Success(List(Abe Allen))

connection
  .retrieveRecords[Person](
    quote { query[Person].filter(_.givenName == "Bee #91232") }
  )
>> 3 mSecs: Success(List(Bee #91232 Worker))
